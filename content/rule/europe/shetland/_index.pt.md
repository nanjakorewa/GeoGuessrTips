    <h2 class="section-title">{{% i18n "how-to-distinguish" %}}</h2>
    <ul class="rule-list">
        <li>Localizadas no extremo norte do Reino Unido{{% ex "https://goo.gl/maps/xYGRhBCP378DnE41A"%}}</li>
